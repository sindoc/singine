\newcommand{\silkbranch}[4]{%
  \ifnum#4>0
    \draw[#1, line width=#2] #3 -- ++(90:#2*16);
    \begin{scope}[shift={#3}]
      \path (90:#2*16) coordinate (tip);
      \draw[#1, line width=#2] (0,0) -- (tip);
      \silkbranch{#1}{#2*0.74}{($(tip)+(132:#2*8)$)}{\numexpr#4-1\relax}
      \silkbranch{#1}{#2*0.74}{($(tip)+(48:#2*8)$)}{\numexpr#4-1\relax}
    \end{scope}
  \fi
}
