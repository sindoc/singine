\newcommand{\fractalpaisleybranch}[5]{%
  \ifnum#5>0
    \paisleycore{#1}{#2}{#3}{#4}
    \paisleydotarc{#1}{#2}{#3}{#4}
    \pgfmathsetmacro{\nextscale}{#3*0.62}
    \pgfmathsetmacro{\xone}{#1 + #3*(0.82*cos(#4) - 2.12*sin(#4))}
    \pgfmathsetmacro{\yone}{#2 + #3*(0.82*sin(#4) + 2.12*cos(#4))}
    \pgfmathsetmacro{\xtwo}{#1 + #3*(0.18*cos(#4) - 4.08*sin(#4))}
    \pgfmathsetmacro{\ytwo}{#2 + #3*(0.18*sin(#4) + 4.08*cos(#4))}
    \pgfmathtruncatemacro{\nextdepth}{#5-1}
    \fractalpaisleybranch{\xone}{\yone}{\nextscale}{#4+18}{\nextdepth}
    \fractalpaisleybranch{\xtwo}{\ytwo}{#3*0.44}{#4-12}{\nextdepth}
  \fi
}
